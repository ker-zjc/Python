for num in range(1, 9):
    filename = "计190" + str(num) + ".csv"#创建以各班级为csv的文件名
    classname = "计190" + str(num)
    with open(filename,"r") as f:
        data = np.loadtxt(f,str,delimiter = ',')
    for a in range(3):#循环三列成绩
        tempmale = np.where(data == '男')#筛选男生
        male_data = data[tempmale[0]]#只取所在行
        male_data_score = male_data[:,3+a].astype(dtype = 'int16')#选第4列的数据，并转换类型
        np.mean(male_data_score)
        tempfemale = np.where(data == '女')#筛选女生
        female_data = data[tempfemale[0]]#只取所在行
        female_data_score = female_data[:,3+a].astype(dtype = 'int16')#选第4列的数据，并转换类型
        np.mean(female_data_score)
        x = ['男生','女生']
        y = [np.mean(male_data_score),np.mean(female_data_score)]
        matplotlib.rcParams['font.sans-serif']=['SimHei']
        plt.title(classname+"男女平均分条柱状图"+data[0,a+3])
        plt.ylabel("分数")
        plt.bar(x, y, alpha=0.5, color='r', edgecolor='k')
        plt.show()

    Fail = 0
    D = 0
    C = 0
    B = 0
    A = 0
    for i in scorelist_name:
        if i < 60:
            Fail += 1
        elif i>= 60 and i < 70:
            D += 1
        elif i>= 70 and i < 80:
            C += 1
        elif i>= 80 and i < 90:
            B += 1
        elif i>= 90 and i <= 100:
            A += 1
    x = ["0-59", "60-69", "70-79", "80-89", "90-100"]
    y = [Fail,D,C,B,A]
    file = open('data_sex.csv', 'w',newline='')
    w = csv.writer(file)
    w.writerow(x)#将列名写入文件第一行
    w.writerow(y)
    file.close()
