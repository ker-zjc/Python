for num in range(1, 9):
    filename = "190" + str(num) + ".csv"#csv
    classname = "190" + str(num)
    with open(filename,"r") as f:
        data = np.loadtxt(f,str,delimiter = ',')
    for a in range(3):#
        tempmale = np.where(data == '')#
        male_data = data[tempmale[0]]#
        male_data_score = male_data[:,3+a].astype(dtype = 'int16')#4
        np.mean(male_data_score)
        tempfemale = np.where(data == '')#
        female_data = data[tempfemale[0]]#
        female_data_score = female_data[:,3+a].astype(dtype = 'int16')#4
        np.mean(female_data_score)
        x = ['','']
        y = [np.mean(male_data_score),np.mean(female_data_score)]
        matplotlib.rcParams['font.sans-serif']=['SimHei']
        plt.title(classname+""+data[0,a+3])
        plt.ylabel("")
        plt.bar(x, y, alpha=0.5, color='r', edgecolor='k')
        plt.show()
